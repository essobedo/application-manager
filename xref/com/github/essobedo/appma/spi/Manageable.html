19  package com.github.essobedo.appma.spi;
20  
21  import com.github.essobedo.appma.exception.ApplicationException;
22  import com.github.essobedo.appma.core.ApplicationManager;
23  import javafx.scene.Scene;
24  import javafx.scene.image.Image;
25  
26  /**
27   * Defines an application that can be managed by the {@link ApplicationManager}.
28   *
29   * @author Nicolas Filotto (nicolas.filotto@gmail.com)
30   * @version $Id$
31   * @since 1.0
32   */
33  public interface Manageable {
34  
35      /**
36       * Gives the name of the application.
37       * @return the name of the application
38       */
39      String name();
40  
41      /**
42       * Gives the identifier of the version of the application.
43       * @return the identifier of the version of the application.
44       */
45      String version();
46  
47      /**
48       * Gives the title of the window to use when launching the application. This is only
49       * needed in case the application is a Java FX application.
50       * @return the title of the window to use when launching the application.
51       */
52      String title();
53  
54      /**
55       * Gives the icon to use when launching the application. This is only
56       * needed in case the application is a Java FX application.
57       * @return the icon to use when launching the application.
58       */
59      Image icon();
60  
61      /**
62       * Indicates whether the application can handle the specified arguments.
63       * @param arguments the arguments to check.
64       * @return {@code true} if the application can handle the specified arguments, {@code false}
65       * otherwise.
66       */
67      boolean accept(String... arguments);
68  
69      /**
70       * Indicates whether the application is a Java FX application.
71       * @return {@code true} if the  application is a Java FX application, {@code false}
72       * otherwise.
73       */
74      boolean isJavaFX();
75  
76      /**
77       * Initializes the application.
78       * @param manager the application manager allowing to interact with it.
79       * @param arguments the arguments to use to initialize the application.
80       * @return a {@link Scene} corresponding to the initialized application if it is a Java FX
81       * application or {@code null} otherwise.
82       * @throws ApplicationException if the application could not be initialized.
83       */
84      Scene init(ApplicationManager manager, String... arguments) throws ApplicationException;
85  
86      /**
87       * Destroys the application.
88       * @throws ApplicationException if the application could not be destroyed.
89       */
90      void destroy() throws ApplicationException;
91  }
