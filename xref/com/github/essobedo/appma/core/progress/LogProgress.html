19  package com.github.essobedo.appma.core.progress;
20  
21  import com.github.essobedo.appma.task.Task;
22  import com.github.essobedo.appma.task.TaskProgress;
23  import java.util.logging.Level;
24  import java.util.logging.Logger;
25  
26  /**
27   * Class allowing to log any progress of a given task. It is used by default for all
28   * non Java FX tasks.
29   *
30   * @author Nicolas Filotto (nicolas.filotto@gmail.com)
31   * @version $Id$
32   * @since 1.0
33   */
34  public final class LogProgress extends TaskProgress {
35  
36      /**
37       * The logger of the class.
38       */
39      private static final Logger LOG = Logger.getLogger(LogProgress.class.getName());
40  
41      /**
42       * Constructs a {@code LogProgress} with the specified task.
43       * @param task the task for which we want to log the progress.
44       */
45      public LogProgress(final Task<?> task) {
46          super(task);
47      }
48  
49      @Override
50      public void updateProgress(final int done, final int max) {
51          if (LOG.isLoggable(Level.INFO)) {
52              LOG.log(Level.INFO, String.format("Task '%s': %d out of %d has been done",
53                  getTask().getName(), done, max));
54          }
55      }
56  
57      @Override
58      public void updateMessage(final String message) {
59          if (LOG.isLoggable(Level.INFO)) {
60              LOG.log(Level.INFO, String.format("Task '%s': %s", getTask().getName(), message));
61          }
62      }
63  
64      @Override
65      public void cancel() {
66          if (LOG.isLoggable(Level.INFO)) {
67              LOG.log(Level.INFO, String.format("The task '%s' has been canceled", getTask().getName()));
68          }
69      }
70  }
