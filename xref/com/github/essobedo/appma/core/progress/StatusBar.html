19  package com.github.essobedo.appma.core.progress;
20  
21  import com.github.essobedo.appma.i18n.Localization;
22  import com.github.essobedo.appma.task.Task;
23  import com.github.essobedo.appma.task.TaskProgressFX;
24  import javafx.geometry.Pos;
25  import javafx.scene.control.Button;
26  import javafx.scene.control.Label;
27  import javafx.scene.control.ProgressBar;
28  import javafx.scene.layout.VBox;
29  
30  /**
31   * Class allowing to show any progress of a given task thanks to a progress bar and a label to display
32   * the status. It is used by default for all Java FX tasks.
33   *
34   * @author Nicolas Filotto (nicolas.filotto@gmail.com)
35   * @version $Id$
36   * @since 1.0
37   */
38  public final class StatusBar extends VBox {
39  
40      /**
41       * The cancel button.
42       */
43      private final Button button;
44  
45      /**
46       * Constructs a {@code StatusBar} with the specified task.
47       * @param task the task for which we want to show the progress.
48       */
49      public StatusBar(final Task<?> task) {
50          super(10);
51          this.setAlignment(Pos.CENTER);
52          final ProgressBar bar = new ProgressBar();
53          bar.setMinWidth(250);
54          final Label label = new Label(task.getName());
55          this.button = new Button(Localization.getMessage("cancel"));
56          button.setOnAction(event -> task.cancel());
57          button.setDisable(!task.cancelable());
58          this.getChildren().addAll(label, bar, button);
59          final Progress progress = new Progress(task);
60          label.textProperty().bind(progress.messageProperty());
61          bar.progressProperty().bind(progress.progressProperty());
62          progress.overProperty().addListener((observable) -> {
63              bar.progressProperty().unbind();
64              bar.setProgress(1d);
65          });
66      }
67  
68      /**
69       * Class allowing to disable the cancel button in case the task has already been canceled.
70       */
71      private class Progress extends TaskProgressFX {
72  
73          /**
74           * Constructs a {@code Progress} with the specified task.
75           * @param task the task for which we want to show the progress.
76           */
77          Progress(final Task<?> task) {
78              super(task);
79          }
80  
81          @Override
82          public void cancel() {
83              StatusBar.this.button.setDisable(true);
84          }
85      }
86  }
