19  package com.github.essobedo.appma.core.zip;
20  
21  import com.github.essobedo.appma.exception.ApplicationException;
22  import com.github.essobedo.appma.task.Task;
23  import com.github.essobedo.appma.i18n.Localization;
24  import java.io.File;
25  import java.io.IOException;
26  
27  /**
28   * {@link Task} allowing to unzip of zip file into a given folder.
29   *
30   * @author Nicolas Filotto (nicolas.filotto@gmail.com)
31   * @version $Id$
32   * @since 1.0
33   */
34  public final class UnzipTask extends Task<Void> {
35  
36      /**
37       * The zip file to unzip.
38       */
39      private final File zipFile;
40      /**
41       * The destination folder.
42       */
43      private final File destFolder;
44  
45      /**
46       * Constructs a {@code UnzipTask} with the specified zip file and destination folder.
47       * @param zipFile the zip file to unzip.
48       * @param destFolder the destination folder.
49       */
50      public UnzipTask(final File zipFile, final File destFolder) {
51          super(Localization.getMessage("patch.unzip"));
52          this.zipFile = zipFile;
53          this.destFolder = destFolder;
54      }
55  
56      @Override
57      public boolean cancelable() {
58          return true;
59      }
60  
61      @Override
62      public Void execute() throws ApplicationException {
63          try {
64              updateMessage(Localization.getMessage("patch.unzipping"));
65              final ZipFile file = new ZipFile(zipFile);
66              file.unzip(destFolder);
67          } catch (IOException e) {
68              throw new ApplicationException("Could not unzip the patch", e);
69          }
70          return null;
71      }
72  }
