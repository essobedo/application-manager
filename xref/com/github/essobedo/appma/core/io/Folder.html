19  package com.github.essobedo.appma.core.io;
20  
21  import java.io.File;
22  import java.io.IOException;
23  import java.nio.file.FileVisitResult;
24  import java.nio.file.Files;
25  import java.nio.file.Path;
26  import java.nio.file.Paths;
27  import java.nio.file.SimpleFileVisitor;
28  import java.nio.file.StandardCopyOption;
29  import java.nio.file.attribute.BasicFileAttributes;
30  import java.util.logging.Level;
31  import java.util.logging.Logger;
32  
33  /**
34   * Class allowing to manipulate a folder and its content.
35   *
36   * @author Nicolas Filotto (nicolas.filotto@gmail.com)
37   * @version $Id$
38   * @since 1.0
39   */
40  public final class Folder {
41  
42      /**
43       * The logger of the class.
44       */
45      private static final Logger LOG = Logger.getLogger(Folder.class.getName());
46  
47      /**
48       * The folder to manipulate.
49       */
50      private final File folderAsFile;
51  
52      /**
53       * Constructs a {@code Folder} using the specified {@code File} as corresponding
54       * folder.
55       * @param folder the folder to manipulate.
56       */
57      public Folder(final File folder) {
58          this.folderAsFile = folder;
59      }
60  
61      /**
62       * Deletes the folder and its content.
63       */
64      public void delete() {
65          try {
66              final Path directory = Paths.get(folderAsFile.getAbsolutePath());
67              Files.walkFileTree(directory, new DeleteFileVisitor());
68          } catch (IOException e) {
69              if (LOG.isLoggable(Level.WARNING)) {
70                  LOG.log(Level.WARNING, String.format("Could not delete the content of the folder '%s'",
71                      folderAsFile.getAbsolutePath()), e);
72              }
73          }
74      }
75  
76      /**
77       * Copies the folder and its content to the specified location.
78       * @param destination the folder in which the content of the folder will be copied
79       */
80      public void copy(final File destination) {
81          try {
82              final Path directory = Paths.get(folderAsFile.getAbsolutePath());
83              final Path target = Paths.get(destination.getAbsolutePath());
84              Files.walkFileTree(directory, new CopyFileVisitor(directory, target));
85          } catch (IOException e) {
86              if (LOG.isLoggable(Level.WARNING)) {
87                  LOG.log(Level.WARNING, String.format("Could not copy the content of the folder '%s'",
88                      folderAsFile.getAbsolutePath()), e);
89              }
90          }
91      }
92  
93      /**
94       * Class allowing to delete the content of the folder.
95       */
96      private static class DeleteFileVisitor extends SimpleFileVisitor<Path> {
97          @Override
98          public FileVisitResult visitFile(final Path file, final BasicFileAttributes attrs) throws IOException {
99              Files.delete(file);
100             return FileVisitResult.CONTINUE;
101         }
102 
103         @Override
104         public FileVisitResult postVisitDirectory(final Path dir, final IOException exc) throws IOException {
105             Files.delete(dir);
106             return FileVisitResult.CONTINUE;
107         }
108     }
109 
110     /**
111      * Class allowing to copy the content of the folder.
112      */
113     private static class CopyFileVisitor extends SimpleFileVisitor<Path> {
114         /**
115          * The source directory.
116          */
117         private final Path directory;
118         /**
119          * The target directory.
120          */
121         private final Path target;
122 
123         /**
124          * Constructs a {@code CopyFileVisitor} using the specified source and target directories.
125          * @param directory the source directory.
126          * @param target the target directory.
127          */
128         CopyFileVisitor(final Path directory, final Path target) {
129             this.directory = directory;
130             this.target = target;
131         }
132 
133         @Override
134         public FileVisitResult visitFile(final Path file, final BasicFileAttributes attrs) throws IOException {
135             Files.copy(file, target.resolve(directory.relativize(file)), StandardCopyOption.REPLACE_EXISTING);
136             return FileVisitResult.CONTINUE;
137         }
138 
139         @Override
140         public FileVisitResult preVisitDirectory(final Path dir, final BasicFileAttributes attrs)
141             throws IOException {
142             Files.copy(dir, target.resolve(directory.relativize(dir)), StandardCopyOption.REPLACE_EXISTING);
143             return FileVisitResult.CONTINUE;
144         }
145     }
146 }
