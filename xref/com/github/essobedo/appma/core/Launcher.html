41  /**
42   * The main entry point to the application manager.
43   *
44   * @author Nicolas Filotto (nicolas.filotto@gmail.com)
45   * @version $Id$
46   * @since 1.0
47   */
48  public final class Launcher extends Application {
49  
50      /**
51       * The logger of the class.
52       */
53      private static final Logger LOG = Logger.getLogger(Launcher.class.getName());
54  
55      /**
56       * The name of the system parameter allowing to externalize the
57       * configuration of the logger.
58       */
59      private static final String PARAM_LOGGER = "java.util.logging.config.file";
60  
61      /**
62       * The current application manager.
63       */
64      private static DefaultApplicationManager applicationManager;
65  
66      /**
67       * The main method.
68       * @param args the arguments.
69       * @throws IOException in case the logger could not be set up.
70       */
71      public static void main(final String... args) throws IOException {
72          setUpLogger();
73  
74          DefaultApplicationManager applicationManager = null;
75          Manageable application = null;
76          try {
77              final RootFolder root = new RootFolder(Launcher.class);
78              applicationManager = new DefaultApplicationManager(root.getLocation(), args);
79              application = applicationManager.create();
80              if (application.isJavaFX()) {
81                  Launcher.applicationManager = applicationManager;
82                  launch(args);
83              } else {
84                  applicationManager.init();
85              }
86          } catch (Exception e) {
87              if (LOG.isLoggable(Level.SEVERE)) {
88                  LOG.log(Level.SEVERE, e.getMessage(), e);
89              }
90              System.exit(1);
91          }
92          if (!application.isJavaFX()) {
93              try {
94                  applicationManager.destroy();
95              } catch (Exception e) {
96                  if (LOG.isLoggable(Level.SEVERE)) {
97                      LOG.log(Level.SEVERE, e.getMessage(), e);
98                  }
99              }
100             System.exit(0);
101         }
102     }
103 
104     /**
105      * Sets up the logger. It will first check if the system property {@code java.util.logging.config.file}
106      * has been set, if so it will let the JDK initialize the logger otherwise it will get the
107      * file {@code /conf/logging.properties} from the classloader.
108      * @throws IOException if the logger could not be set up.
109      */
110     private static void setUpLogger() throws IOException {
111         final File logs = new File("logs");
112         if (logs.exists() || logs.mkdir()) {
113             if (System.getProperty(Launcher.PARAM_LOGGER) != null) {
114                 // Let the JDK configure the logger
115                 return;
116             }
117             final LogManager manager = LogManager.getLogManager();
118             try (final InputStream input = Launcher.class.getResourceAsStream("/conf/logging.properties")) {
119                 manager.readConfiguration(input);
120             }
121         }
122     }
123 
124     @Override
125     public void start(final Stage primaryStage) {
126         start(Launcher.applicationManager, primaryStage);
127     }
128 
129     /**
130      * Starts the launcher using the specified application manager.
131      * @param applicationManager the application manager to use.
132      * @param primaryStage the primary stage for this application, onto which
133      * the application scene can be set. The primary stage will be embedded in
134      * the browser if the application was launched as an applet.
135      * Applications may create other stages, if needed, but they will not be
136      * primary stages and will not be embedded in the browser.
137      * @return The {@link Future} object allowing to be notified once the application has
138      * been initialized and shown.
139      */
140     static Future<Void> start(final DefaultApplicationManager applicationManager, final Stage primaryStage) {
141         final VBox vBox = new VBox(10);
142         vBox.setAlignment(Pos.CENTER);
143         final ProgressBar bar = new ProgressBar();
144         bar.setMinWidth(250);
145         final Label label = new Label(Localization.getMessage("status.loading"));
146         final Button button = new Button(Localization.getMessage("close"));
147         button.setOnAction((event) -> applicationManager.onExit());
148         button.setDisable(true);
149         vBox.getChildren().addAll(label, bar, button);
150         primaryStage.setScene(new Scene(vBox, 300.0d, 150.0d));
151         final Manageable application = applicationManager.getApplication();
152         primaryStage.setResizable(false);
153         primaryStage.setOnCloseRequest(Event::consume);
154         if (application.title() == null) {
155             primaryStage.setTitle(Localization.getMessage("title.window"));
156         } else {
157             primaryStage.setTitle(application.title());
158         }
159         if (application.icon() != null) {
160             primaryStage.getIcons().add(application.icon());
161         }
162         primaryStage.show();
163 
164         return applicationManager.asyncInitNShow(primaryStage,
165             () -> {
166                 label.setText(Localization.getMessage("status.error"));
167                 button.setDisable(false);
168             }
169         );
170     }
