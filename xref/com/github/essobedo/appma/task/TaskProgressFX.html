19  package com.github.essobedo.appma.task;
20  
21  import com.github.essobedo.appma.exception.ApplicationException;
22  import com.github.essobedo.appma.exception.TaskInterruptedException;
23  import javafx.application.Platform;
24  import javafx.beans.property.BooleanProperty;
25  import javafx.beans.property.DoubleProperty;
26  import javafx.beans.property.ReadOnlyBooleanProperty;
27  import javafx.beans.property.ReadOnlyDoubleProperty;
28  import javafx.beans.property.ReadOnlyStringProperty;
29  import javafx.beans.property.SimpleBooleanProperty;
30  import javafx.beans.property.SimpleDoubleProperty;
31  import javafx.beans.property.SimpleStringProperty;
32  import javafx.beans.property.StringProperty;
33  
34  /**
35   * The root class of the {@code TaskObserver} in case of a Java FX application.
36   *
37   * @author Nicolas Filotto (nicolas.filotto@gmail.com)
38   * @version $Id$
39   * @since 1.0
40   * @param <T> The return type of the task.
41   */
42  @SuppressWarnings("PMD.AbstractNaming")
43  public abstract class TaskProgressFX<T> extends TaskProgress {
44  
45      /**
46       * The progress of the task.
47       */
48      private final DoubleProperty progress = new SimpleDoubleProperty(this, "progress", -1);
49      /**
50       * The status of the task.
51       */
52      private final StringProperty message = new SimpleStringProperty(this, "message", "");
53      /**
54       * Indicates whether the task is over.
55       */
56      private final BooleanProperty over = new SimpleBooleanProperty(this, "over", false);
57  
58      /**
59       * Constructs a {@code TaskProgressFX} with the specified task.
60       * @param task the task to observe.
61       */
62      public TaskProgressFX(final Task<T> task) {
63          super(task);
64      }
65  
66      /**
67       * Gives the current progress of the task.
68       * @return the current progress of the task.
69       */
70      public final double getProgress() {
71          return progress.get();
72      }
73      /**
74       * Gives the current progress of the task.
75       * @return the current progress of the task.
76       */
77      public final ReadOnlyDoubleProperty progressProperty() {
78          return progress;
79      }
80      /**
81       * Gives the current status of the task.
82       * @return the current status of the task.
83       */
84      public final String getMessage() {
85          return message.get();
86      }
87      /**
88       * Gives the current status of the task.
89       * @return the current status of the task.
90       */
91      public final ReadOnlyStringProperty messageProperty() {
92          return message;
93      }
94      /**
95       * Indicates whether the task is over.
96       * @return {@code true} if the task is over, {@code false} otherwise.
97       */
98      public final boolean isOver() {
99          return over.get();
100     }
101     /**
102      * Indicates whether the task is over.
103      * @return {@code true} if the task is over, {@code false} otherwise.
104      */
105     public final ReadOnlyBooleanProperty overProperty() {
106         return over;
107     }
108     /**
109      * Cancels the underlying task if possible.
110      */
111     public final void cancelTask() {
112         getTask().cancel();
113     }
114 
115     @Override
116     public final void updateProgress(final int done, final int max) {
117         Platform.runLater(() -> {
118             if (done <= max && max > 0) {
119                 this.progress.set((double) (done / max));
120             } else {
121                 this.progress.set(-1.0d);
122             }
123         });
124     }
125 
126     @Override
127     public final void updateMessage(final String message) {
128         Platform.runLater(() -> this.message.setValue(message));
129     }
130 
131     /**
132      * Executes the underlying task.
133      * @return the result of the task.
134      * @throws ApplicationException if an error occurs while executing the task.
135      * @throws TaskInterruptedException if the task has been interrupted.
136      */
137     public final T execute() throws ApplicationException, TaskInterruptedException {
138         try {
139             return (T) getTask().execute();
140         } finally {
141             Platform.runLater(() -> this.over.setValue(true));
142         }
143     }
144 }
