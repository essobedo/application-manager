19  package com.github.essobedo.appma.task;
20  
21  import java.util.Observable;
22  import java.util.Observer;
23  
24  /**
25   * Defines the {@link Observer} of a {@link Task}.
26   *
27   * @author Nicolas Filotto (nicolas.filotto@gmail.com)
28   * @version $Id$
29   * @since 1.0
30   */
31  public interface TaskObserver extends Observer {
32  
33      @Override
34      default void update(final Observable observable, final Object object) {
35          final Task<?> task;
36          if (observable instanceof Task) {
37              task = (Task<?>) observable;
38          } else {
39              throw new IllegalArgumentException("A task was expected");
40          }
41          final Task.Event event;
42          if (object instanceof Task.Event) {
43              event = (Task.Event) object;
44          } else {
45              throw new IllegalArgumentException("A task event was expected");
46          }
47          switch (event) {
48              case PROGRESS:
49                  updateProgress(task.getWorkDone(), task.getMax());
50                  break;
51              case MESSAGE:
52                  updateMessage(task.getMessage());
53                  break;
54              case CANCEL:
55                  cancel();
56                  break;
57              default:
58                  throw new IllegalArgumentException("Unknown event");
59          }
60      }
61  
62      /**
63       * Notifies that the current progress of the task has changed.
64       * @param done the work already done.
65       * @param max the maximum work to be done.
66       */
67      void updateProgress(int done, int max);
68  
69      /**
70       * Notifies that the status of the task has changed.
71       * @param message the status of the task.
72       */
73      void updateMessage(String message);
74  
75      /**
76       * Notifies that the task has been canceled.
77       */
78      void cancel();
79  }
