19  package com.github.essobedo.appma.core;
20  
21  import com.github.essobedo.appma.core.DefaultApplicationManager;
22  import com.github.essobedo.appma.exception.ApplicationException;
23  import java.util.concurrent.CountDownLatch;
24  import java.util.concurrent.ExecutionException;
25  import javafx.scene.Scene;
26  import javafx.scene.layout.VBox;
27  import javafx.stage.Stage;
28  import org.junit.Test;
29  import org.testfx.framework.junit.ApplicationTest;
30  import static org.junit.Assert.assertTrue;
31  import static org.junit.Assert.fail;
32  
33  /**
34   * @author Nicolas Filotto (nicolas.filotto@gmail.com)
35   * @version $Id$
36   * @since 1.0
37   */
38  public class TestFXStates extends ApplicationTest {
39      private final String folderName = "app.upgrade.fx.ok3";
40  
41      private DefaultApplicationManager manager;
42  
43      public void stop() {
44          try {
45              manager.destroy();
46          } catch (ApplicationException e) {
47              throw new IllegalStateException(e);
48          }
49      }
50  
51      @Test
52      public void testStates() throws Exception {
53          CountDownLatch latch = new CountDownLatch(1);
54          try {
55              manager.asyncInitNShow(null, latch::countDown).get();
56              fail("An ApplicationException is expected");
57          } catch (ExecutionException e) {
58              assertTrue(e.getCause() instanceof ApplicationException);
59          }
60          latch.await();
61      }
62  
63      @Override
64      public void start(final Stage stage) throws Exception {
65          this.manager = new DefaultApplicationManager(
66              TestDefaultApplicationManager.getRootFolder(folderName));
67          manager.create();
68          Scene scene = new Scene(new VBox(), 200.0d, 150.0d);
69          stage.setScene(scene);
70          stage.show();
71          manager.init();
72      }
73  }
