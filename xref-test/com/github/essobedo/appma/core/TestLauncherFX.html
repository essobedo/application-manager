35  import static org.testfx.matcher.base.NodeMatchers.hasText;

37  /**
38   * @author Nicolas Filotto (nicolas.filotto@gmail.com)
39   * @version $Id$
40   * @since 1.0
41   */
42  public class TestLauncherFX extends ApplicationTest implements ChangeListener<Scene> {
43      private CountDownLatch latch = new CountDownLatch(2);
44      private final String folderName = "app.upgrade.fx.ok2";
45      private DefaultApplicationManager manager;
46  
47      public void init() throws Exception {
48          this.manager = new DefaultApplicationManager(
49              TestDefaultApplicationManager.getRootFolder(folderName));
50          manager.create();
51      }
52  
53      public void stop() {
54          manager.getStage().sceneProperty().removeListener(this);
55          try {
56              manager.destroy();
57          } catch (ApplicationException e) {
58              throw new IllegalStateException(e);
59          }
60      }
61  
62      @Test
63      public void testAppFXUpgrade() throws Exception {
64          latch.await();
65          latch = new CountDownLatch(1);
66          Manageable application = manager.getApplication();
67          assertEquals("foo FX", application.name());
68          assertEquals("1.0", application.version());
69          assertEquals("title FooFX", application.title());
70          assertTrue(application.isJavaFX());
71          verifyThat("#label", hasText("foo"));
72          ApplicationManager applicationManager = manager;
73          try {
74              System.setProperty("test.folder", folderName);
75              assertEquals("2.0", applicationManager.checkForUpdate().execute());
76              manager.doUpgrade();
77          } finally {
78              System.clearProperty("test.folder");
79          }
80          latch.await();
81      }
82  
83      @Test
84      public void testAppFXUpgraded() throws Exception {
85          latch.await();
86          Manageable application = manager.getApplication();
87          assertNotNull(application);
88          assertEquals("foo FX", application.name());
89          assertEquals("2.0", application.version());
90          assertEquals("title FooFX", application.title());
91          assertTrue(application.isJavaFX());
92          verifyThat("#label", hasText("bar"));
93      }
94  
95      @Override
96      public void start(final Stage stage) throws Exception {
97          stage.sceneProperty().addListener(this);
98          Launcher.start(manager, stage);
99      }
100 
101     @Override
102     public void changed(final ObservableValue<? extends Scene> observable, final Scene oldValue, final Scene newValue) {
103         latch.countDown();
104     }
105 }
