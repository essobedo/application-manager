19  package com.github.essobedo.appma.core.io;
20  
21  import java.io.File;
22  import java.nio.file.Files;
23  import java.nio.file.Path;
24  import org.junit.Test;
25  import static org.junit.Assert.assertEquals;
26  import static org.junit.Assert.assertFalse;
27  import static org.junit.Assert.assertTrue;
28  
29  /**
30   * @author Nicolas Filotto (nicolas.filotto@gmail.com)
31   * @version $Id$
32   * @since 1.0
33   */
34  public class TestFolder {
35  
36      @Test
37      public void testActions() throws Exception {
38          Path tempDir = Files.createTempDirectory("TestFolder");
39          Folder folder = new Folder(new File(TestFolder.class.getResource("/folder").toURI()));
40          File target = new File(tempDir.toString());
41          folder.copy(target);
42          assertEquals(2, target.list().length);
43          assertTrue(new File(target, "subfolder1").exists());
44          assertTrue(new File(target, "subfolder1").isDirectory());
45          assertTrue(new File(target, "subfolder1/foo.properties").exists());
46          assertTrue(new File(target, "subfolder1/foo.properties").isFile());
47          assertTrue(new File(target, "test.properties").exists());
48          assertTrue(new File(target, "test.properties").isFile());
49  
50          assertTrue(target.exists());
51          folder = new Folder(target);
52          folder.delete();
53          assertFalse(target.exists());
54      }
55  
56      @Test
57      public void testActionsFailure() throws Exception {
58          Path tempDir = Files.createTempDirectory("TestFolder");
59          File dir = new File(tempDir.toString());
60          Folder folder = new Folder(dir);
61          assertTrue(dir.delete());
62          assertFalse(dir.exists());
63          folder.delete();
64          folder.copy(new File("foo"));
65      }
66  }
