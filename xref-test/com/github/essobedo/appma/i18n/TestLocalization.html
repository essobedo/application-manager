19  package com.github.essobedo.appma.i18n;
20  
21  import org.junit.Test;
22  import static org.junit.Assert.assertEquals;
23  import static org.junit.Assert.fail;
24  
25  /**
26   * @author Nicolas Filotto (nicolas.filotto@gmail.com)
27   * @version $Id$
28   * @since 1.0
29   */
30  public class TestLocalization {
31  
32      @Test
33      public void testConstructor() {
34          try {
35              new Localization("foo");
36              fail("A RuntimeException");
37          } catch (RuntimeException e) {
38              // expected
39          }
40          new Localization("i18n.foo");
41      }
42  
43      @Test
44      public void testLocalizedMessage() {
45          Localization localization = new Localization("i18n.foo");
46          assertEquals("Hello", localization.getLocalizedMessage("key1"));
47          assertEquals("Hello World!", localization.getLocalizedMessage("key2", "World"));
48          assertEquals("key3", localization.getLocalizedMessage("key3", "World"));
49          assertEquals("key4", localization.getLocalizedMessage("key4"));
50      }
51  }
